CREATE DATABASE dbProducts
USE dbProducts

CREATE TABLE tUser(
UserId INT CONSTRAINT PK_User PRIMARY KEY IDENTITY(1,1),
IsAdmin INT,
FirstName VARCHAR(20) NOT NULL,
LastName VARCHAR(20) NOT NULL,
PhoneNumber VARCHAR(20) NOT NULL,
Password VARCHAR(20)
)

CREATE TABLE tProduct(
ProductId INT CONSTRAINT PK_Product PRIMARY KEY IDENTITY(1,1),
Description VARCHAR(20) NOT NULL,
Price int NOT NULL,
DateManufactured DATETIME NOT NULL,
)

CREATE TABLE tCategory(
CategoryId INT CONSTRAINT PK_Category PRIMARY KEY IDENTITY(1,1),
CategoryName VARCHAR(20)
)

CREATE TABLE tProductCategory(
ProductId INT FOREIGN KEY REFERENCES tProduct(ProductId),
CategoryId INT FOREIGN KEY REFERENCES tCategory(CategoryId)
)

CREATE TABLE tProductEvaluation(
ProductId INT FOREIGN KEY REFERENCES tProduct(ProductId),
UserId INT FOREIGN KEY REFERENCES tUser(UserId),
PRIMARY KEY (ProductId,UserId),
Value INT
)
